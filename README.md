# install
 clone project

```bash
 git clone https://github.com/yuri7772d/roomQ.git
 
 cd roomQ

```
 create file .env
```bash
PORT=81
JWT_SECRET=ddffaasdf
JWT_REFRESH_SECRET=dfadsadasf
ROOT_USERNAME=admin
ROOT_PASSWORD=1234
MYSQL_HOST=localhost
MYSQL_PORT=3333
MYSQL_DB=roomq
MYSQL_USERNAME=sql
MYSQL_PASSWORD=1234

```
 install & run node.js

```bash
 #install node
 npm install

 #run dev
 npm run dev

 #run test
 npm run test
```
